// 1.Написать функцию, которая возвращает произведение двух целых чисел,
// полученных в качестве аргумента.
int Prod(int a, int b)
{
    if (a && b != 0)
    {
        return a * b;
    }
    return 0;
}

{
    int a = 0, b = 0, res = 0;
    cout << "Enter two numbers ti product ==> ";
    cin >> a >> b;
    res = Prod(a, b);
    cout << res;
}

// 2. Написать функцию, которая возвращает количество цифр в целом числе,
// полученном в качестве аргумента.
int Counter(int num)
{
    int res = 0;
    for (int i = num; i > 0; i = i / 10)
    {
        res++;
    }
    return res;
}

int main()
{
    int num = 0, res = 0;
    cout << "Enter number ==> ";
    cin >> num;
    res = Counter(num);
    cout << "There are(is) " << res << " digits";
}

// сравнения в виде одного из знаков: >, < или =.
int Comparison(int a, int b)
{
    char res;
    (a > b) ? res = '>' : (a < b) ? res = '<' : res = '=';
    return res;
}

{
